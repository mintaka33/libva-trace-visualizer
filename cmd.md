## FFmpeg

#### generate vide stream
```bash
python3 vis.py "ffmpeg -f lavfi -i testsrc2 -vframes 100 test.264"
```

#### decode
```bash
python3 vis.py "ffmpeg -loglevel verbose -hwaccel vaapi -i test.264 -f null -"
```

#### decode single thread
```bash
python3 vis.py "ffmpeg -loglevel verbose -threads 1 -hwaccel vaapi -i test.264 -f null -"
```

#### transcode 1:1 AVC -> AVC
```bash
python3 vis.py "ffmpeg -hwaccel vaapi -vaapi_device /dev/dri/renderD128 -v verbose -hwaccel_output_format vaapi -i test.264 -vf scale_vaapi=w=300:h=200 -c:v h264_vaapi -vframes 100 -an -y /tmp/out.264"
```

#### transcode 1:1 AVC -> HEVC
```bash
python3 vis.py "ffmpeg -hwaccel vaapi -vaapi_device /dev/dri/renderD128 -v verbose -hwaccel_output_format vaapi -i test.264 -vf scale_vaapi=w=300:h=200 -c:v hevc_vaapi -vframes 100 -an -y /tmp/out.265"
```

#### transcode 1:4
```bash
python3 vis.py "ffmpeg -hwaccel vaapi -vaapi_device /dev/dri/renderD128 -v verbose -hwaccel_output_format vaapi -i test.264 -vf scale_vaapi=w=160:h=120 -c:v hevc_vaapi -vframes 100 -an -y /tmp/out1.265 -vf scale_vaapi=w=240:h=160 -c:v hevc_vaapi -vframes 100 -an -y /tmp/out2.265 -vf scale_vaapi=w=320:h=240 -c:v hevc_vaapi -vframes 100 -an -y /tmp/out3.265 -vf scale_vaapi=w=480:h=320 -c:v hevc_vaapi -vframes 100 -an -y /tmp/out4.265"
```

#### h264 decode
```bash
python3 vis.py "gst-launch-1.0 filesrc location=/home/fresh/data/video/test.mp4 ! decodebin ! fakesink"

python3 vis.py "gst-launch-1.0 filesrc location=/home/fresh/data/video/test.mp4 ! qtdemux ! h264parse ! vaapih264dec ! fakesink sync=false"

python3 vis.py "gst-launch-1.0 filesrc location=/home/fresh/data/video/test.mp4 ! qtdemux ! h264parse ! vaapih264dec ! gvafpscounter ! fakesink sync=false"
```

#### decode + VPP
```bash
python3 vis.py "gst-launch-1.0 filesrc location=/home/fresh/data/video/test.mp4 ! qtdemux ! h264parse ! vaapih264dec ! vaapipostproc ! videoconvert ! video/x-raw, format=BGR ! gvafpscounter ! fakesink sync=false"

python3 vis.py "gst-launch-1.0 filesrc location=/home/fresh/data/video/test2.mp4 ! qtdemux ! h264parse ! vaapih264dec ! 
vaapipostproc ! videoconvert ! video/x-raw, format=BGR ! filesink location=out.yuv"
```

#### 1:1 transcode
```bash
python3 vis.py "gst-launch-1.0 filesrc location=/home/fresh/data/video/test.mp4 ! qtdemux ! h264parse ! vaapih264dec ! vaapipostproc ! video/x-raw,format=NV12,width=300,height=300 ! vaapih265enc rate-control=cbr bitrate=1500 ! gvafpscounter ! filesink location=/tmp/out.265"
```
